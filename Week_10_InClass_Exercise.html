Stationarity Testing — ADF Test

Apply the Augmented Dickey-Fuller test before and after transformation.

🔬 2A — ADF Test on Raw Data

PYTHON 
def adf_test(series, name="Series"):
    result = adfuller(series.dropna())
    print(f"\n{'='*45}")
    print(f"  ADF Test — {name}")
    print(f"{'='*45}")
    print(f"  ADF Statistic : {result[0]:.4f}")
    print(f"  p-value       : {result[1]:.4f}")
    print(f"  Critical Values: 1%={result[4]['1%']:.3f}, 5%={result[4]['5%']:.3f}")
    if result[1] < 0.05:
        print("  ✅ STATIONARY  (p < 0.05 → Reject H₀)")
    else:
        print("  ❌ NON-STATIONARY  (p ≥ 0.05 → Fail to Reject H₀)")

# Test original series
adf_test(df["Passengers"], "Original Passengers")

PYTHON 
# 1. Log transformation (stabilizes variance)
df["Log_Pass"] = np.log(df["Passengers"])

# 2. First-order differencing (removes trend)
df["Diff1"] = df["Passengers"].diff()

# 3. Log + first-order differencing (removes both)
df["Log_Diff1"] = df["Log_Pass"].diff()

# Run ADF on all transformations
adf_test(df["Log_Pass"],   "Log(Passengers)")
adf_test(df["Diff1"],      "First Difference")
adf_test(df["Log_Diff1"],  "Log + First Difference")

📏 Complete Evaluation Function

PYTHON 
def evaluate(actual, predicted, model_name):
    mae  = mean_absolute_error(actual, predicted)
    rmse = np.sqrt(mean_squared_error(actual, predicted))
    mape = np.mean(np.abs((actual.values - predicted.values) / actual.values)) * 100
    r2   = r2_score(actual, predicted)
    return {"Model": model_name, "MAE": round(mae,2), "RMSE": round(rmse,2),
            "MAPE(%)": round(mape,2), "R²": round(r2,4)}

results = [
    evaluate(test, ses_fc,    "SES (α=0.3)"),
    evaluate(test, des_fc,    "Holt's DES"),
    evaluate(test, hw_fc,     "Holt-Winters"),
    evaluate(test, arima_fc,  "ARIMA(2,1,1)"),
    evaluate(test, sarima_fc, "SARIMA(1,1,1)(1,1,1,12)"),
]

results_df = pd.DataFrame(results).set_index("Model")
